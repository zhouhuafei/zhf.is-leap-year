# 检测是否是闰年
```
const isLeapYear = require('zhf.is-leap-year');

console.log(isLeapYear(2000)); // true
console.log(isLeapYear(2012)); // true
console.log(isLeapYear(2013)); // false
console.log(isLeapYear(2014)); // false
console.log(isLeapYear(2015)); // false
console.log(isLeapYear(2016)); // true
console.log(isLeapYear(2017)); // false
console.log(isLeapYear(2018)); // false
```
